>url<"E:/sem6/data.csv"//url where your csv file is located this is for ex
>input<-read.csv(url)
>head(input)
//after that the table is shown as index Height and Weight that are headers
>attach(input)
>relationWeightHeight<lm(Weight~Height)
>plot(input$Height, input$Weight,
+ main="Height vs Weight",
+ xlab="Height",
+ ylab="Weight",
+ pch=16
+ col="blue")

>abline(relationWeightHeight, col="red",lwd=2)
>predicted<-predict)relationWeightHeight,input)
>head(predicted)
>coef(relationWeightHeight)

>plot(input$Height, input$Weight,
+main ="Height vs Weight with Regression Line",
+ xlab ="Height",
+ ylab ="Weight",
+ pch=16
+ col ="blue")

>
>lines(input$Height, predicted,
+ col="red",
